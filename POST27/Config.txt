ASA Configuration
enable

configure terminal

interface ethernet0/0
 nameif outside
 security-level 0
 ip address 200.1.1.1 255.255.255.0
 no shutdown

interface ethernet0/1
 nameif inside
 security-level 100
 ip address 192.168.1.1 255.255.255.0
 no shutdown

end
